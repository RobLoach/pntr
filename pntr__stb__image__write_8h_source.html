102  pntr_stb_image_context context;
103  context.data = NULL;

105 
106  switch (type) {
107  case PNTR_IMAGE_TYPE_UNKNOWN:
108  case PNTR_IMAGE_TYPE_PNG: {

116  break;
117  case PNTR_IMAGE_TYPE_BMP: {

124  }
